数据结构学习——队列

2021-03-01 Java 数据结构学习

  1. 同栈一样，队列（queue）也是表。然而使用队列时插入在一端进行删除在另一端进行。区别于栈的后进先出，队列是先进先出。
  2. 队列的基本操作有入队（enqueue），即在队尾（rear）插入一个元素；出队（dequeue），即删除队头（front）元素。
  3. 数组模拟实现队列。

class QueueImpl<T> implements MyQueue<T>{
    private T[] data;
    private int size;
    private int front;
    private int rear;
    public QueueImpl() {
        data = (T[])new Object[3];
        size = 0;
        front = 0;
        rear = 0;
    }

    @Override
    public void add(T t) {
        if(isFull()){
            resize();
            front = 0;
        }
        rear = (front+size)%data.length;
        data[rear] = t;
        size++;
    }

    @Override
    public T remove() {
        if(isEmpty()){
            throw new RuntimeException("queue is empty!");
        }
        T tempData = data[front];
        data[front] = null;
        front = (front+1)%data.length;
        size--;
        return tempData;
    }

    @Override
    public int size() {
        return size;
    }

    @Override
    public boolean isEmpty() {
        return size == 0;
    }

    @Override
    public T front() {
        if(isEmpty()){
            throw new RuntimeException("queue is empty!");
        }
        return data[front];
    }

    public boolean isFull(){
        return size == data.length;
    }

    public void resize(){
        T[] tmp = (T[])new Object[data.length*2];
        System.arraycopy(data, 0, tmp, 0, data.length);
        data = tmp;
        tmp = null;
    }

}

interface MyQueue<T> {
    void add(T t);
    T remove();
    int size();
    boolean isEmpty();
    T front();
}
